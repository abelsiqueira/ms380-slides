\myframetop{
  \begin{center}
  \begin{tikzpicture}[scale=0.7]
    \draw[thick] (0,0) -- node[below] {$b$} (4,0) -- 
      node[right] {$c$} (4,3) -- node[above] {$a$} (0,0) {};
    \only<1,3>{\draw (1,0) arc (0:36.9:1) 
      node[midway,right] {$\alpha$};}
    \only<2>{\draw (4,2) arc (-90:-90-53.1:1) 
      node[midway,below] {$\theta$};}
    \draw (3.5,0) -- (3.5, 0.5) -- (4,0.5) {};
  \end{tikzpicture}
  \end{center}
  \only<1>{\begin{align*}
    \sin\alpha & = \frac{c}{a} \\
    \cos\alpha & = \frac{b}{a} \\
    \tan\alpha & = \frac{c}{b} = \frac{\sin\alpha}{\cos\alpha}
  \end{align*}}
  \only<2>{\begin{align*}
    \theta & = \frac{\pi}{2} - \alpha \\
    \sin\theta & = \frac{b}{a} \\
    \cos\theta & = \frac{c}{a} \\
    \tan\theta & = \frac{b}{c} = \frac{\sin\theta}{\cos\theta}
  \end{align*}}
  \only<3>{\begin{align*}
    \sin(\pi/2-\alpha) & = \cos(\alpha) \\
    \cos(\pi/2-\alpha) & = \sin(\alpha) \\
    \tan(\pi/2-\alpha) & = \frac{1}{\tan(\alpha)} \\
  \end{align*}}
}

    \cos(a+b) & = \cos(a)\cos(b) - \sin(a)\sin(b) \\
